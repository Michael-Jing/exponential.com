    fn union_fn(g: &mut Vec, s: &mut Vec, i: usize, j: usize) {
        let i = Self::find(g, i);
        let j = Self::find(g, j);
        if i != j {
            if s[j] > s[i] {
                g[i] = j;
                s[j] += s[i];
            } else {
                g[j] = i;
                s[i] += s[j];
            }
        }
    }
